add_namespace = alcoy

# Alcoy se une al bando republicano durante la Guerra Civil
country_event = {
	id = alcoy.5
	title = alcoy.5.t
	desc = alcoy.5.d
	picture = GFX_report_event_spanish_civil_war_soldiers
	
	is_triggered_only = no
	fire_only_once = yes
	
	trigger = {
		tag = ALC
		has_global_flag = spanish_civil_war
		is_subject_of = SPD
	}
	
	immediate = {
		hidden_effect = {
			if = {
				limit = { is_subject_of = SPD }
				SPD = { end_puppet = ALC }
			}
		}
	}
	
	mean_time_to_happen = {
		days = 1
	}
	
	option = {
		name = alcoy.5.a
		ai_chance = { factor = 100 }
		
		if = {
			limit = { SPR = { has_government = democratic } }
			SPR = { add_to_faction = ALC }
		}
		else_if = {
			limit = { D01 = { exists = yes } }
			D01 = { add_to_faction = ALC }
		}
		else = {
			SPR = { add_to_faction = ALC }
		}
		
		add_war_support = 0.15
		add_stability = -0.05
		add_political_power = 50
		
		custom_effect_tooltip = alcoy_joins_republicans_tooltip
	}
}
