# Francia - M치s propensa a enviar voluntarios
country_event = {
	id = alcoy_confederation.1
	title = alcoy_confederation.1.t
	desc = alcoy_confederation.1.d
	picture = GFX_report_event_spanish_civil_war_soldiers
	
	is_triggered_only = yes
	
	# Enviar voluntarios
	option = {
		name = alcoy_confederation.1.a
		ai_chance = { factor = 50 }
		
		send_volunteers_to = ALC
		add_political_power = -50
		add_opinion_modifier = {
			target = ALC
			modifier = large_increase
		}
	}
	
	# Enviar equipamiento
	option = {
		name = alcoy_confederation.1.b
		ai_chance = { factor = 30 }
		
		ALC = {
			add_equipment_to_stockpile = {
				type = infantry_equipment
				amount = 2000
			}
			add_equipment_to_stockpile = {
				type = artillery_equipment
				amount = 200
			}
		}
		add_political_power = -25
		add_opinion_modifier = {
			target = ALC
			modifier = medium_increase
		}
	}
	
	# Enviar manpower
	option = {
		name = alcoy_confederation.1.c
		ai_chance = { factor = 20 }
		
		ALC = {
			add_manpower = 5000
		}
		add_manpower = -5000
		add_political_power = -25
		add_opinion_modifier = {
			target = ALC
			modifier = medium_increase
		}
	}
	
	# No hacer nada
	option = {
		name = alcoy_confederation.1.e
		ai_chance = { factor = 10 }
		add_political_power = 25
	}
}

# Evento de victoria y paz scripteada
country_event = {
	id = alcoy_confederation.100
	title = alcoy_confederation.100.t
	desc = alcoy_confederation.100.d
	picture = GFX_report_event_generic_sign_treaty2
	
	fire_only_once = yes
	is_triggered_only = no
	
	trigger = {
		has_global_flag = mediterranean_confederation_war
		SPR = {
			OR = {
				has_capitulated = yes
				surrender_progress > 0.8
				exists = no
			}
		}
	}
	
	mean_time_to_happen = {
		days = 1
	}
	
	immediate = {
		# 游댮 FIN REAL DE LA GUERRA (ejecuta SOLO el vencedor)
		ALC = {
			end_war = {
				target = SPR
				result = victory
			}
		}
	}
	
	option = {
		name = alcoy_confederation.100.a
		
		# ===============================
		# REDISTRIBUCI칍N TERRITORIAL
		# ===============================
		
		# Murcia obtiene Albacete (1052)
		if = {
			limit = { MUR = { exists = yes } }
			MUR = { transfer_state = 1052 }
		}
		
		# Alcoy obtiene estado 783
		transfer_state = 783
		
		# ===============================
		# CREACI칍N DE NUEVOS ESTADOS
		# ===============================
		
		# Arag칩n (166, 794)
		release = ARA
		ARA = {
			transfer_state = 166
			transfer_state = 794
			add_state_core = 166
			add_state_core = 794
			set_politics = {
				ruling_party = democratic
				elections_allowed = yes
			}
		}
		add_to_faction = ARA
		
		# Pa칤s Vasco (792, 172)
		release = NAV
		NAV = {
			transfer_state = 792
			transfer_state = 172
			add_state_core = 792
			add_state_core = 172
			set_politics = {
				ruling_party = democratic
				elections_allowed = yes
			}
		}
		add_to_faction = NAV
		
		# Granada (173)
		release = ADU
		ADU = {
			transfer_state = 173
			add_state_core = 173
			set_politics = {
				ruling_party = democratic
				elections_allowed = yes
			}
		}
		add_to_faction = ADU
		
		# Galicia (171)
		release = GLC
		GLC = {
			transfer_state = 171
			add_state_core = 171
			set_politics = {
				ruling_party = democratic
				elections_allowed = yes
			}
		}
		add_to_faction = GLC
		
		# S치hara Occidental (699)
		release = WES
		WES = {
			transfer_state = 699
			add_state_core = 699
			set_politics = {
				ruling_party = neutrality
				elections_allowed = no
			}
		}
		add_to_faction = WES
		
		# Guinea Ecuatorial (297)
		release = EQG
		EQG = {
			transfer_state = 297
			add_state_core = 297
			set_politics = {
				ruling_party = democratic
				elections_allowed = yes
			}
		}
		add_to_faction = EQG
		
		# ===============================
		# ZONAS DESMILITARIZADAS
		# ===============================
		
		169 = { set_demilitarized_zone = yes } # Sevilla
		789 = { set_demilitarized_zone = yes } # C칩rdoba
		175 = { set_demilitarized_zone = yes } # Ciudad Real
		793 = { set_demilitarized_zone = yes } # Guadalajara
		176 = { set_demilitarized_zone = yes } # Burgos
		790 = { set_demilitarized_zone = yes } # Asturias
		174 = { set_demilitarized_zone = yes } # Le칩n
		
		# ===============================
		# POSTGUERRA - Espa침a desaparece, nace Corona de Castilla
		# ===============================
		
		# CST hereda todos los estados restantes de Espa침a
		every_state = {
			limit = {
				is_owned_by = SPR
			}
			CST = { transfer_state = PREV }
		}
		
		# Espa침a es anexada (desaparece)
		every_state = {
			limit = {
				is_core_of = SPR
			}
			add_core_of = CST
		}
		CST = {
			annex_country = {
				target = SPR
				transfer_troops = yes
			}
		}
		
		clr_global_flag = mediterranean_confederation_war
		set_global_flag = mediterranean_confederation_victory
		
		add_political_power = 200
		add_stability = 0.15
	}
}
