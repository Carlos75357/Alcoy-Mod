# ============================================================================
# Alcoy puede aumentar su influencia sobre los países de la confederación
# reduciendo su estabilidad. Cuando estabilidad <= 15%, puede vasallizarlos.
# Los países pueden resistir aumentando su estabilidad.
# Si estabilidad >= 90%, la acción se cancela.
# ============================================================================

alc_more_influence_decisions = {

	# Decisión dinámica: Aumentar influencia (reducir estabilidad del aliado)
	alc_increase_influence = {
		icon = GFX_decision_generic_propaganda
		
		target_trigger = {
			FROM = {
				is_in_faction_with = ROOT
				NOT = { tag = ROOT }
				NOT = { is_subject_of = ROOT }
				has_stability > 0.15
			}
		}

		visible = {
			has_country_flag = more_influence
			FROM = {
				is_in_faction_with = ROOT
				NOT = { is_subject_of = ROOT }
			}
		}
